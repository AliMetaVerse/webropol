Event

Fires webropol-button-hue-click (CustomEvent) with detail: { hue, orientation, label } on click.

JavaScript API

// Listen for click events
document.querySelector('webropol-button-hue')
  .addEventListener('webropol-button-hue-click', (e) => {
    console.log(e.detail); // { hue, orientation, label }
  });

// Programmatic disable
const btn = document.querySelector('webropol-button-hue');
btn.setDisabled(true);
btn.setDisabled(false);
